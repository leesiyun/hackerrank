const timeConversion = (s) => {
    let answer = "";
    s.slice(-2) === "AM" &&
      (s.slice(0, 2) === "12"
        ? (answer = "00" + s.slice(2, 8))
        : (answer = s.slice(0, 8)));
    s.slice(-2) === "PM" &&
      (s.slice(0, 2) === "12"
        ? (answer = "12" + s.slice(2, 8))
        : (answer = Number(s.slice(0, 2)) + 12 + s.slice(2, 8)));
    return answer;
};
