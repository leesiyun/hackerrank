## solution

```JS
const miniMaxSum = (arr) => {
    arr.sort();
    let min = arr.slice(0, 4).reduce((acc, cur) => acc + cur);
    let max = arr.slice(1, 5).reduce((acc, cur) => acc + cur);
    return console.log(`${min} ${max}`);
};
```
