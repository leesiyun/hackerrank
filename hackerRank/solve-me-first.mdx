## solution

```JS
const solveMeFirst = (a, b) => a + b;
const constraintCheck = (a, b) => a >= 1 && a <= 1000 && b >= 1 && b <= 1000;

function main() {
    var a = parseInt(readLine());
    var b = parseInt(readLine());

    const res = constraintCheck(a, b)
      ? solveMeFirst(a, b)
      : "constraint violated";

    console.log(res);
}
```
