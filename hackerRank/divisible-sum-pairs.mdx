## solution

```JS
const divisibleSumPairs = (n, k, ar) => {
    let count = 0;
    ar.forEach((x, i) => {
      while (i <= n) {
        i++;
        if ((x + ar[i]) % k === 0) count++;
      }
    });
    return count;
};
```
